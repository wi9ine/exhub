```json PrivateV2.1 && PublicV2
{
    "result": "error",
    "error_code": "105",
    "error_msg": "Price is not correct"
}
```
```json PrviateV2 && PublicV1
{
    "result": "error",
    "errorCode": "105",
    "errorMsg": "Price is not correct"
}
```
